Map with draggable Markers, custom icon and infobox, no streetviewcontrol, no maptypecontrol

Note: only the arrow-marker is draggable

Code:

// create the map:
$('#canvas').goomaps({
	center: [52.257347,0.054988],
	zoom: 6,
	mapTypeId: 'roadmap',
	streetViewControl: false,
	mapTypeControl: false
});

// add some markers:
markers = [
	{
		options: {
			position: [52.257347,0.054988],
			title: 'A Test Marker',
			clickable: true,
			draggable: true,
			info: '#info',
			icon: 'http://maps.google.com/mapfiles/arrow.png',
			shadow: 'http://maps.google.com/mapfiles/arrowshadow.png',
		},
		events: {
			'dragend': function(event){
				alert(this.getTitle() + ': ' + this.getPosition());
			}
		}
	},
	{
		options: {
			position: [52, 10],
			title: 'Another Test Marker',
			draggable: false,
			info: 'Another Test Marker'
		}
	}
];
$('#canvas').goomaps('setmarkers', markers);
