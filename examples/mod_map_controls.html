Map with modified Map-Controls

Positioning and styling the navigation-controls and maptype-controls

Code:

$('#canvas').goomaps({
	center: [52.257347,0.054988],
	zoom: 6,
	type: 'roadmap',
	navigationControlOptions: {
		position: "bottom",
		style: "android"
	},
	mapTypeControlOptions: {
		position: "left",
		style: "dropdown_menu"
	}
});
