Map with modified Map-Controls

Positioning and styling the Google Maps Controls

Code:

$('#canvas').goomaps({
	center: [52.257347,0.054988],
	zoom: 6,
	type: 'roadmap',
	zoomControlOptions: {
		position: "right_center"
	},
	mapTypeControlOptions: {
		position: "top_left",
		style: "dropdown_menu"
	}
});
