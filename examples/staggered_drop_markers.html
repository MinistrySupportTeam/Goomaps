Drop markers onto map with staggered timeout

Using setevents and setTimeout to drop markers on to the map

Code:

	var markers = [
		{
			"options" => {
				"uid" => "longstanton",

				"draggable" => true,
				"info" => "Longstanton Marker",
				"animation" => "drop"
			}
		},

	$(document).ready(function() {
		$('#canvas').goomaps({
			"center":[52.26017,0.0683],
			"MapTypeId":"roadmap",
			"zoom":10
		});
		$.goomaps.setevents($('#goomap').goomaps('getmap'),{
			'tilesloaded': function(ev){
				$.each(results, function(i, result){
					setTimeout(function(){
						$('#goomap').goomaps('addmarkers', result);
					}, (i + 1) * 200);
				});
			}
		}, 'once');
	});
